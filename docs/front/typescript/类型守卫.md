# 类型守卫

* 多态 - 

```ts
interface Sale {
    name: string,
    speak: string[]
}
interface Front {
    name: string,
    code: string
}

type Class = Sale | Front

// 1、in
function work(cls: Class) {
    if("speak" in cls) {
      console('sale')  
    }
    if("code" in cls) {
        console('Front')
    }
}

// 2、typeof 
function quit(name: string, reason: any[] | string) {
    if(typeof reason === 'array') {
        console('sale')
    }
    if(typeof reason === 'string') {
        console('Front')
    }
}

// 2、instanceof 
function getPros(cls: Class) {
    if(cls instanceof Sale) {
        console('sale')
    }
    if(cls instanceof Front) {
        console('Front')
    }
}

```
